            class Solution {
            public int compress(char[] chars) {
                char prev = chars[0];
                int count = 1;
                StringBuilder answer = new StringBuilder();
                
                for(int i=1 ; i<chars.length ; i++){
                    char ch = chars[i];
                    if(prev != ch){
                        answer.append(prev);
                        if(count > 1)answer.append(count);
                        count = 1;
                        prev = ch;
                    }
                    else{
                        count++;
                    }
                }

                answer.append(prev);
                if(count > 1) answer.append(count);
                
                for(int i=0 ; i<answer.length() ; i++){
                    chars[i] = answer.charAt(i);
                }
                return answer.length();
            }
        }
