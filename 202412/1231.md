            class Solution {
            public int maximumUnits(int[][] boxTypes, int truckSize) {
                Arrays.sort(boxTypes, (o1, o2) -> o2[1] - o1[1]);

                int answer = 0, size = 0;
                for(int[] boxType : boxTypes){
                    if(size + boxType[0] < truckSize){
                        answer += boxType[0] * boxType[1];
                        size += boxType[0];
                    }
                    else{
                        answer += (truckSize - size) * boxType[1];
                        break;
                    }
                }

                return answer;
            }
        }
